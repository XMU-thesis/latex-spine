\documentclass[UTF8]{ctexbook}

% https://stackoverflow.com/a/2458125/13182099
\def\hyphenateWholeString #1{\xHyphenate#1$\wholeString\unskip}

\def\xHyphenate#1#2\wholeString {\if#1$
\else\transform{#1}
\takeTheRest#2\ofTheString\fi}
\def\takeTheRest#1\ofTheString\fi
{\fi \xHyphenate#1\wholeString}

\def\transform#1{#1\par}

\renewcommand{\title}[1]{\def\spinetitle{\hyphenateWholeString{#1}}}
\renewcommand{\author}[1]{\def\spineauthor{\hyphenateWholeString{#1}}}
\newcommand{\adviser}[3]{\def\spineadviser{\hyphenateWholeString{#1}~\par\hyphenateWholeString{#2}~\par\hyphenateWholeString{#3}}}
\newcommand{\institute}[1]{\def\spineinst{\hyphenateWholeString{#1}}}


\newcommand{\makespine}{

    \newpage\thispagestyle{empty}
    \linespread{1.0}
    \songti\zihao{-5}\centering
    \spinetitle\vfill\spineauthor\vfill\spineadviser\vfill\spineinst

}
